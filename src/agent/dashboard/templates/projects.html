Quản lý qua API

# Tạo project mới
curl -X POST localhost:8000/projects \
  -H "Content-Type: application/json" \
  -d '{"id":"my-project","name":"My Project","description":"..."}'

# Thêm service
curl -X POST localhost:8000/projects/my-project/services \
  -H "Content-Type: application/json" \
  -d '{"service":"payment-gateway"}'

# Chọn LLM riêng cho project
curl -X PATCH localhost:8000/projects/my-project/llm \
  -H "Content-Type: application/json" \
  -d '{"provider":"gemini","model":"gemini-2.0-flash"}'
